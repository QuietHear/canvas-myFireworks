# 参数配置
>
    g: 0.25, // 模拟重力加速度，默认0.25
    n: 5, // 每次发射的烟花的数量，默认5
    surprise_num: 25, // 烟花总量，默认25
    flag: true, // 是否发射生日快乐字样的烟火，默认true
    infinite: true, // 是否循环，默认false
>
